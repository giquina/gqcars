## Step 2: Google Apps Script Automation

### ⚡ **Auto-Document Creation Script:**
```javascript
function createAIResearchReport(data) {
  // Template document ID
  var templateId = 'your-template-document-id';
  
  // Create copy of template
  var template = DriveApp.getFileById(templateId);
  var newDoc = template.makeCopy('AI Research - ' + data.query + ' - ' + new Date().toDateString());
  
  // Open the document for editing
  var doc = DocumentApp.openById(newDoc.getId());
  var body = doc.getBody();
  
  // Replace placeholders with actual data
  body.replaceText('\\[AUTO-FILLED\\]', new Date().toLocaleString());
  body.replaceText('\\[CLAUDE RESPONSE\\]', data.claudeResponse);
  body.replaceText('\\[CHATGPT RESPONSE\\]', data.chatgptResponse);
  body.replaceText('\\[PERPLEXITY RESPONSE\\]', data.perplexityResponse);
  body.replaceText('\\[AUTO-CALCULATED\\]', data.similarityScore);
  body.replaceText('\\[AI-GENERATED\\]', data.aiSummary);
  
  // Save and close
  doc.saveAndClose();
  
  // Move to appropriate folder
  var folder = DriveApp.getFolderById('your-reports-folder-id');
  folder.addFile(newDoc);
  DriveApp.getRootFolder().removeFile(newDoc);
  
  return newDoc.getUrl();
}
```

function insertChart(docId, chartData) {
  var doc = DocumentApp.openById(docId);
  var body = doc.getBody();
  
  // Create chart from Sheets data
  var chart = Charts.newColumnChart()
    .setTitle('AI Performance Comparison')
    .setXAxisTitle('AI Models')
    .setYAxisTitle('Response Time (ms)')
    .addRange(chartData)
    .build();
  
  // Insert chart into document
  body.appendImage(chart);
}

### 🎨 **Smart Formatting:**
```javascript
function formatDocument(docId) {
  var doc = DocumentApp.openById(docId);
  var body = doc.getBody();
  
  // Format headings
  var headings = body.findText('EXECUTIVE SUMMARY|CLAUDE ANALYSIS|CHATGPT ANALYSIS');
  while (headings) {
    headings.getElement().asText().setFontSize(14).setBold(true);
    headings = body.findText('EXECUTIVE SUMMARY|CLAUDE ANALYSIS|CHATGPT ANALYSIS', headings);
  }
  
  // Format action items
  var actionItems = body.findText('•.*');
  while (actionItems) {
    actionItems.getElement().asText().setForegroundColor('#0066cc');
    actionItems = body.findText('•.*', actionItems);
  }
}
```

### 💬 **Add AI Comments:**

function addAIInsights(docId, insights) {
  var doc = DocumentApp.openById(docId);
  var body = doc.getBody();
  
  // Find key sections and add AI-generated comments
  var summary = body.findText('EXECUTIVE SUMMARY');
  if (summary) {
    doc.addComment(summary.getElement().asRange(), 
      '🤖 AI Insight: ' + insights.executiveSummaryTip);
  }
}
